### 前端工具函数

**utils/member-limit.js**
```javascript
/**
 * 检查是否可以添加植物
 */
function canAddPlant() {
  return new Promise(async (resolve) => {
    const isVip = getApp().globalData.vipUser || false
    
    if (isVip) {
      resolve({ canAdd: true, isVip: true })
      return
    }
    
    // 非会员检查数量限制
    const db = wx.cloud.database()
    const countRes = await db.collection('my_plants').count()
    
    if (countRes.total >= 5) {
      resolve({
        canAdd: false,
        isVip: false,
        currentCount: countRes.total,
        maxCount: 5,
        message: '非会员最多只能添加 5 盆植物'
      })
    } else {
      resolve({
        canAdd: true,
        isVip: false,
        remainingCount: 5 - countRes.total
      })
    }
  })
}

/**
 * 检查是否可以识别植物
 */
function canIdentify() {
  return new Promise((resolve) => {
    const isVip = getApp().globalData.vipUser || false
    
    if (isVip) {
      resolve({ canIdentify: true, isVip: true })
      return
    }
    
    // 非会员检查每日次数
    const today = new Date().toISOString().split('T')[0]
    const identifyHistory = getApp().globalData.identifyHistory || []
    const todayRecords = identifyHistory.filter(r => r.date === today && r.success)
    
    if (todayRecords.length >= 10) {
      resolve({
        canIdentify: false,
        isVip: false,
        todayCount: todayRecords.length,
        maxCount: 10,
        message: '非会员每天只能识别 10 次植物'
      })
    } else {
      resolve({
        canIdentify: true,
        isVip: false,
        remainingCount: 10 - todayRecords.length
      })
    }
  })
}

/**
 * 显示会员限制提示
 */
function showLimitAlert(type) {
  if (type === 'plant') {
    wx.showModal({
      title: '植物数量已达上限',
      content: '非会员最多只能添加 5 盆植物。\n\n开通会员即可添加无限盆植物！',
      confirmText: '开通会员',
      cancelText: '再看看',
      success: (res) => {
        if (res.confirm) {
          wx.navigateTo({ url: '/pages/vip/vip' })
        }
      }
    })
  } else if (type === 'identify') {
    wx.showModal({
      title: '今日识别次数已达上限',
      content: '非会员每天只能识别 10 次植物。\n\n开通会员即可无限次识别！',
      confirmText: '开通会员',
      cancelText: '明天再来',
      success: (res) => {
        if (res.confirm) {
          wx.navigateTo({ url: '/pages/vip/vip' })
        }
      }
    })
  }
}
```

## 📊 数据埋点

### 关键转化漏斗

1. **曝光 → 点击**：VIP 卡片曝光数 / 点击进入 VIP 页面数
2. **点击 → 支付**：VIP 页面访问数 / 发起支付数
3. **支付 → 成功**：发起支付数 / 支付成功数

### 核心指标

| 指标 | 目标值 | 计算方式 |
|------|--------|---------|
| 付费转化率 | 3-5% | 付费用户数 / 总用户数 |
| 月 ARPPU | ¥15-20 | 月收入 / 付费用户数 |
| 续费率 | 40-60% | 续费用户数 / 到期用户数 |
| LTV | ¥60-100 | 用户生命周期总价值 |

## ⚠️ 注意事项

### 小程序审核

1. **虚拟支付限制**：iOS 端不能直接显示支付按钮（需走虚拟支付报备）
2. **价格展示**：必须明确标注价格单位（元/月、元/年）
3. **自动续费**：必须明确提示用户，并提供取消入口
4. **退款政策**：需在协议中明确退款规则

### 合规要求

1. **会员服务协议**：必须用户勾选同意
2. **隐私政策**：明确说明会员数据使用方式
3. **发票开具**：提供电子发票开具入口
